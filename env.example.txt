# Supabase 配置示例文件
# 复制此文件内容到你的环境变量配置中
# 在 uni-app 中，可以通过条件编译或配置文件管理

# Supabase 项目 URL
# 在 Supabase 项目设置中可以找到
VUE_APP_SUPABASE_URL=https://your-project.supabase.co

# Supabase Anon/Public Key
# 在 Supabase 项目设置 > API 中可以找到
VUE_APP_SUPABASE_ANON_KEY=your-anon-key-here

# 可选：Service Role Key（仅用于服务端，不要在前端使用）
